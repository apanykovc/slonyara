# Explicitly pin pydantic and its core runtime to versions that provide
# pre-built wheels for Python 3.13 on Windows so that Rust is not required
# during installation.  Without these pins pip may attempt to build
# pydantic-core from source which fails on systems without a Rust toolchain.
#
# aiogram 3.4.1 currently depends on pydantic <2.6, so we must stay within
# that range to avoid resolver conflicts while still selecting versions with
# wheels for the supported platforms.
pydantic==2.5.3
pydantic-core==2.14.6
